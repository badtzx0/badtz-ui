  ```tsx showLineNumbers
  import { ClassValue, clsx } from "clsx";
  import { twMerge } from "tailwind-merge";

  export function cn(...inputs: ClassValue[]) {
    return twMerge(clsx(inputs));
  }
  ```

## Usage

```tsx showLineNumbers
import { AnimatedGradient } from "@/components/ui/stripe-animated-gradient";
```

```tsx showLineNumbers
<AnimatedGradient
  color1="#a960ee"
  color2="#ff333d"
  color3="#90e0ff"
  color4="#ffcb57"
/>
```
