## Usage

```tsx showLineNumbers
import { ImageTrail } from "@/components/ui/image-trail";
```

```tsx showLineNumbers
const imageUrls = [
  "/images/components/image-trail/image-trail-1.webp",
  "/images/components/image-trail/image-trail-2.webp",
  "/images/components/image-trail/image-trail-3.webp",
  "/images/components/image-trail/image-trail-4.webp",
  "/images/components/image-trail/image-trail-5.webp",
];
```

```tsx showLineNumbers
<ImageTrail images={imageUrls} imageHeight={150} imageWidth={150} />
```

## Props

`image-trail` props.

<TypeTable
  type={{
    images: {
      type: "string[]",
      description: "An array of image URLs to be used in the trail effect.",
      default: "[]"
    },
    imageWidth: {
      type: "number",
      description: "The width of each image in the trail (in pixels).",

    },

      description: "The height of each image in the trail (in pixels).",

    },
    threshold: {
      type: "number",
      description: "The minimum distance the mouse must move to trigger an image change.",
      default: "50"
    },
    duration: {
      type: "number",
      description: "The duration (in seconds) of the animation for each image.",

    }
  }}
