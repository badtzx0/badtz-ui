```jsx
import { cn } from "@/lib/utils";

export function Marquee({
  children,
  direction = "left",
  repeat = 4,
  duration = 60,
  className,
  ...props
}) {
  return (
    <div
      {...props}
      className={cn(
        "group flex overflow-hidden [--gap:1rem] [gap:var(--gap)]",
        className
      )}
      style={{ "--duration": `${duration}s` }}
    >
      {Array(repeat)
        .fill(0)
        .map((_, i) => (
          <div
            key={i}
            className={cn("flex shrink-0 justify-around [gap:var(--gap)]", {
              "animate-marquee-left": direction === "left",
              "animate-marquee-right": direction === "right",
              "group-hover:[animation-play-state:paused]": true,
            })}
          >
            {children}
          </div>
        ))}
    </div>
  );
}
```

<Step>Update `tailwind.config.js`</Step>

Add the following animations to your `tailwind.config.js` file:

```js {5-20}
/** @type {import('tailwindcss').Config} */
module.exports = {
  theme: {
    extend: {
      keyframes: {
        "marquee-left": {
          from: { transform: "translateX(0)" },
          to: { transform: "translateX(calc(-100% - var(--gap)))" },
        },
        "marquee-right": {
          from: { transform: "translateX(calc(-100% - var(--gap)))" },
          to: { transform: "translateX(0)" },
        },
      },
      animation: {
        "marquee-left": "marquee-left var(--duration, 30s) linear infinite",
        "marquee-right": "marquee-right var(--duration, 30s) linear infinite",
      },
    },
  },
};
```

## Props

`marquee` props.

| Prop      | Type            | Description                                            | Default |
| --------- | --------------- | ------------------------------------------------------ | ------- |
| direction | "left" \| "right" | Direction of the scrolling animation.                 | "left"  |
| repeat    | number          | Number of times the content is repeated.              | 4       |
| duration  | number          | Duration of the animation in seconds.                 | 60      |
| children  | React.ReactNode | The content to be displayed in the marquee.           | -       |
| className | string          | Additional custom classes for styling the marquee.    | -       |
