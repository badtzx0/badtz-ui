## Usage

```tsx showLineNumbers
import { ExpandableCard } from "@/components/ui/expandable-card";
```

```tsx showLineNumbers
<ExpandableCard
  title="Whispering Forest"
  src="/images/components/expandable-card/haunted-house.webp"
  description="A Yokai Tale"
>
  {content}
</ExpandableCard>
```
