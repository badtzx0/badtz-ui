## Usage

```tsx showLineNumbers
import { AnimatedList } from "@/components/ui/animated-list";
```

```tsx showLineNumbers
<AnimatedList>
  {notifications.map((notification) => (
    <Notification key={notification.id} notification={notification} />
  ))}
</AnimatedList>
```
